//CURSOR CREATION

create procedure cur
is
cursor jas is select ename,sal,hiredate from emp;
A varchar(50);
B number(10,2);
C date;
begin
open jas;
loop
fetch jas into A,B,C;
exit when jas%notfound;
dbms_output.put_line(A||' '||B||' '||C);
end loop;
close jas;
end;
/

create procedure join
is
cursor jin is select emp.ename,dept.loc  from emp,dept where emp.deptno=dept.deptno;
A varchar(50);
B varchar(40);
begin
open jin;
loop
fetch jin into A,B;
exit when jin%notfound;
dbms_output.put_line(A||' '||B);
end loop;
close jin;
end;
/
